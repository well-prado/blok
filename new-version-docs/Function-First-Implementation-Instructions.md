## 1. FnNodeDefinition + defineNode

Put this in something like `core/nodes/defineNode.ts` (or equivalent shared package) so it can be imported by node implementations. It wraps a function‑first definition into a `BlokService` subclass that matches the existing Node contract (constructor + `handle(ctx, inputs)` and `BlokResponse` usage). [17][27][24][32]

```ts
// core/nodes/defineNode.ts

import { BlokService, BlokResponse, type IBlokResponse } from "@blok/runner"; // BlokService template + response type [17]
import { type Context, GlobalError } from "@blok/shared"; // Context + GlobalError used in existing Node template [17]
import type { ZodSchema } from "zod";
import { ZodError } from "zod";

/**
 * Function-first Node definition:
 * - name: logical node name (used for logging, errors, etc.)
 * - input: Zod schema for inputs
 * - output: Zod schema for outputs
 * - execute: core business logic using Context + validated input
 */
export interface FnNodeDefinition<I, O> {
	name: string;
	description?: string;
	input: ZodSchema<I>;
	output: ZodSchema<O>;
	execute: (ctx: Context, input: I) => Promise<O> | O;
}

/**
 * defineNode:
 * Wraps a function-first node definition into a BlokService-compatible instance.
 * This allows registering it in the standard `nodes: { [key: string]: NodeBase }` maps. [27][18][11]
 */
export function defineNode<I, O>(def: FnNodeDefinition<I, O>) {
	class FunctionNode extends BlokService<I> {
		constructor() {
			super();
			// Keep schemas present for compatibility; can be enhanced later to generate JSON Schema from Zod. [17][27]
			this.inputSchema = {};
			this.outputSchema = {};
			this.name = def.name;
			// If NodeBase / BlokService exposes description, keep it; otherwise this is safe no-op metadata. [17][29][31]
			(this as any).description = def.description ?? (this as any).description;
		}

		async handle(ctx: Context, rawInputs: I): Promise<IBlokResponse> {
			const response: BlokResponse = new BlokResponse(); // same pattern as template Node. [17]

			try {
				// 1) Validate and coerce input using Zod
				const parsed = def.input.parse(rawInputs);

				// 2) Run function-first business logic
				const result = await def.execute(ctx, parsed);

				// 3) Validate output
				const parsedOutput = def.output.parse(result);

				response.setSuccess(parsedOutput); // standard success path. [17]
			} catch (err: unknown) {
				const message = err instanceof Error ? err.message : "Unknown error";
				const nodeError: GlobalError = new GlobalError(message); // same pattern as class template. [17]
				nodeError.setCode(500);
				nodeError.setStack(err instanceof Error ? err.stack : undefined);
				nodeError.setName(this.name);
				nodeError.setJson(
					err instanceof ZodError ? { issues: err.issues } : undefined,
				);

				response.setError(nodeError); // standard error path. [17]
			}

			return response;
		}
	}

	// Return instance so it can be put directly into the nodes map (no `new` at registration). [27][18][14]
	return new FunctionNode();
}
```

This is intentionally aligned with the existing class template generated by the CLI (constructor sets `inputSchema`/`outputSchema`, `handle` creates `BlokResponse`, uses `GlobalError`, and returns `IBlokResponse`). [17][27][24]
